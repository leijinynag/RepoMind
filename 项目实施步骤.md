#### 1.3 实现 RepoManager
**`server/src/repo/RepoManager.ts`：**
```typescript
import simpleGit from 'simple-git';
import path from 'path';
import fs from 'fs/promises';
import { v4 as uuidv4 } from 'uuid';
import { Repo } from '../models/Repo.model';

export class RepoManager {
  private reposDir: string;

  constructor() {
    this.reposDir = path.resolve(process.cwd(), process.env.REPOS_DIR || '../repos');
  }

  async cloneRepo(url: string): Promise<string> {
    // 验证 GitHub URL
    if (!url.match(/^https:\/\/github\.com\/[\w-]+\/[\w-]+/)) {
      throw new Error('Invalid GitHub URL');
    }

    const repoId = uuidv4();
    const repoName = url.split('/').pop()?.replace('.git', '') || 'unknown';
    const localPath = path.join(this.reposDir, repoId);

    // 创建 repo 记录
    const repo = new Repo({
      repoId,
      url,
      name: repoName,
      status: 'cloning',
      localPath,
    });
    await repo.save();

    try {
      // 确保目录存在
      await fs.mkdir(localPath, { recursive: true });

      // clone（shallow clone 节省时间）
      const git = simpleGit();
      await git.clone(url, localPath, ['--depth', '1']);

      // 统计文件数
      const fileCount = await this.countFiles(localPath);

      // 更新状态
      repo.status = 'ready';
      repo.fileCount = fileCount;
      await repo.save();

      return repoId;
    } catch (error: any) {
      repo.status = 'error';
      repo.error = error.message;
      await repo.save();
      throw error;
    }
  }

  private async countFiles(dir: string): Promise<number> {
    let count = 0;
    const files = await fs.readdir(dir, { withFileTypes: true });

    for (const file of files) {
      if (file.name === '.git') continue;
      if (file.name.startsWith('.')) continue;

      const fullPath = path.join(dir, file.name);
      if (file.isDirectory()) {
        count += await this.countFiles(fullPath);
      } else {
        count++;
      }
    }
    return count;
  }

  async getRepo(repoId: string) {
    return Repo.findOne({ repoId });
  }

  async listRepos() {
    return Repo.find().sort({ createdAt: -1 });
  }

  async deleteRepo(repoId: string) {
    const repo = await Repo.findOne({ repoId });
    if (!repo) throw new Error('Repo not found');

    // 删除本地文件
    await fs.rm(repo.localPath, { recursive: true, force: true });

    // 删除数据库记录
    await Repo.deleteOne({ repoId });
  }
}
```

**`server/src/agent/clients/DeepSeekClient.ts`：**
```typescript
import axios from 'axios';
import { LLMClient, Message } from '../LLMClient';

export class DeepSeekClient extends LLMClient {
  private apiKey: string;
  private baseURL = 'https://api.deepseek.com/v1';

  constructor(apiKey: string) {
    super();
    this.apiKey = apiKey;
  }

  async chat(messages: Message[]): Promise<string> {
    const response = await axios.post(
      `${this.baseURL}/chat/completions`,
      {
        model: 'deepseek-chat',
        messages,
        temperature: 0.7,
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
      }
    );

    return response.data.choices[0].message.content;
  }

  async stream(messages: Message[], onToken: (token: string) => void): Promise<void> {
    const response = await axios.post(
      `${this.baseURL}/chat/completions`,
      {
        model: 'deepseek-chat',
        messages,
        temperature: 0.7,
        stream: true,
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        responseType: 'stream',
      }
    );

    return new Promise((resolve, reject) => {
      response.data.on('data', (chunk: Buffer) => {
        const lines = chunk.toString().split('\n').filter(line => line.trim());
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.slice(6);
            if (data === '[DONE]') continue;
            try {
              const parsed = JSON.parse(data);
              const token = parsed.choices[0]?.delta?.content;
              if (token) onToken(token);
            } catch (e) {}
          }
        }
      });

      response.data.on('end', resolve);
      response.data.on('error', reject);
    });
  }
}
```

#### 2.2 实现 Tool 系统基础
**`server/src/tools/ToolRegistry.ts`：**
```typescript
export interface ToolSchema {
  name: string;
  description: string;
  parameters: {
    [key: string]: {
      type: string;
      description: string;
      required?: boolean;
    };
  };
}

export interface Tool {
  name: string;
  schema: ToolSchema;
  execute(input: Record<string, any>, context: { repoId: string }): Promise<string>;
}

export class ToolRegistry {
  private tools: Map<string, Tool> = new Map();

  register(tool: Tool) {
    this.tools.set(tool.name, tool);
  }

  getSchemas(): ToolSchema[] {
    return Array.from(this.tools.values()).map(t => t.schema);
  }

  async execute(name: string, input: Record<string, any>, context: { repoId: string }): Promise<string> {
    const tool = this.tools.get(name);
    if (!tool) {
      throw new Error(`Unknown tool: ${name}`);
    }
    return tool.execute(input, context);
  }

  has(name: string): boolean {
    return this.tools.has(name);
  }
}
```

```typescript
import fs from 'fs/promises';
import path from 'path';
import { Tool, ToolSchema } from './ToolRegistry';
import { Repo } from '../models/Repo.model';

export const readFileTool: Tool = {
  name: 'read_file',
  schema: {
    name: 'read_file',
    description: '读取指定文件的完整内容',
    parameters: {
      file_path: {
        type: 'string',
        description: '文件相对路径，例如 "src/index.ts"',
        required: true,
      },
    },
  },

  async execute(input, context) {
    const { file_path } = input;
    if (!file_path) {
      throw new Error('file_path is required');
    }

    const repo = await Repo.findOne({ repoId: context.repoId });
    if (!repo) {
      throw new Error('Repo not found');
    }

    const fullPath = path.join(repo.localPath, file_path);

    // 安全检查：防止路径穿越
    if (!fullPath.startsWith(repo.localPath)) {
      throw new Error('Invalid file path');
    }

    try {
      const content = await fs.readFile(fullPath, 'utf-8');
      return `File: ${file_path}\n\n${content}`;
    } catch (error: any) {
      return `Error reading file: ${error.message}`;
    }
  },
```

```typescript
export const listFilesTool: Tool = {
  name: 'list_files',
  schema: {
    name: 'list_files',
    description: '列出指定目录下的文件和子目录',
    parameters: {
      dir_path: {
        type: 'string',
        description: '目录路径，留空表示根目录',

    const repo = await Repo.findOne({ repoId: context.repoId });
```

#### 2.5 实现 ReAct 输出解析
**`server/src/agent/parseReActOutput.ts`：**
```typescript
export interface ReActParsed {
  type: 'action' | 'final_answer';
  thought?: string;
  action?: string;
  actionInput?: Record<string, any>;
  finalAnswer?: string;
}

export function parseReActOutput(text: string): ReActParsed {
  // 提取 Thought
  const thoughtMatch = text.match(/Thought:\s*(.+?)(?=\n(?:Action|Final Answer):|$)/s);
  const thought = thoughtMatch?.[1]?.trim();

  // 检查是否是 Final Answer
  const finalAnswerMatch = text.match(/Final Answer:\s*(.+)/s);
  if (finalAnswerMatch) {
    return {
      type: 'final_answer',
      thought,
      finalAnswer: finalAnswerMatch[1].trim(),
    };
  }

  // 提取 Action
  const actionMatch = text.match(/Action:\s*(\w+)/);
  const action = actionMatch?.[1];

  // 提取 Action Input（JSON 格式）
  const inputMatch = text.match(/Action Input:\s*(\{.+?\})/s);
  let actionInput: Record<string, any> = {};
  if (inputMatch) {
    try {
      actionInput = JSON.parse(inputMatch[1]);
    } catch (e) {
      // 如果不是 JSON，尝试简单键值对
      const simpleMatch = text.match(/Action Input:\s*(.+)/);
      if (simpleMatch) {
        actionInput = { query: simpleMatch[1].trim() };
      }
    }
  }

  return {
    type: 'action',
    thought,
    action,
    actionInput,
  };
}
```

#### 2.7 实现 AgentRunner（核心）
**`server/src/agent/AgentRunner.ts`：**
```typescript
import { LLMClient, Message } from './LLMClient';
import { ToolRegistry } from '../tools/ToolRegistry';
import { PromptBuilder } from './PromptBuilder';
import { parseReActOutput } from './parseReActOutput';

export interface AgentEvent {
  type: 'thought' | 'action' | 'observation' | 'answer' | 'error';
  data: any;
}

export class AgentRunner {
  private llm: LLMClient;
  private tools: ToolRegistry;
  private maxSteps = 8;

  constructor(llm: LLMClient, tools: ToolRegistry) {
    this.llm = llm;
    this.tools = tools;
  }

  async run(
    question: string,
    repoId: string,
    repoName: string,
    onEvent: (event: AgentEvent) => void
  ): Promise<void> {
    const systemPrompt = PromptBuilder.buildSystemPrompt(repoName, this.tools.getSchemas());

    const messages: Message[] = [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: question },
    ];

    for (let step = 0; step < this.maxSteps; step++) {
      try {
        // LLM 推理
        const response = await this.llm.chat(messages);
        const parsed = parseReActOutput(response);

        // 发送 thought 事件
        if (parsed.thought) {
          onEvent({ type: 'thought', data: parsed.thought });
        }

        // 如果是最终答案
        if (parsed.type === 'final_answer') {
          onEvent({ type: 'answer', data: parsed.finalAnswer });
          break;
        }

        // 如果是 action
        if (parsed.type === 'action' && parsed.action) {
          onEvent({
            type: 'action',
            data: { tool: parsed.action, input: parsed.actionInput },
          });

          // 执行工具
          if (!this.tools.has(parsed.action)) {
            const error = `Unknown tool: ${parsed.action}`;
            onEvent({ type: 'observation', data: error });
            messages.push({ role: 'assistant', content: response });
            messages.push({ role: 'user', content: `Observation: ${error}` });
            continue;
          }

          const result = await this.tools.execute(parsed.action, parsed.actionInput || {}, { repoId });

          onEvent({ type: 'observation', data: result });

          // 更新对话历史
          messages.push({ role: 'assistant', content: response });
          messages.push({ role: 'user', content: `Observation: ${result}` });
        } else {
          // 解析失败，提示重新格式化
          messages.push({ role: 'assistant', content: response });
          messages.push({
            role: 'user',
            content: 'Please follow the format: Thought -> Action -> Action Input or Final Answer',
          });
        }
      } catch (error: any) {
        onEvent({ type: 'error', data: error.message });
        break;
      }
    }

    // 超过最大步数
    if (messages.length >= this.maxSteps * 2) {
      onEvent({ type: 'error', data: 'Max steps reached' });
    }
  }
}
```

#### 2.8 创建 Chat API（SSE 流式）
**`server/src/api/chat.routes.ts`：**
```typescript
import { Router } from 'express';
import { createLLMClient } from '../agent/clients';
import { ToolRegistry } from '../tools/ToolRegistry';
import { readFileTool } from '../tools/readFile';
import { listFilesTool } from '../tools/listFiles';
import { AgentRunner } from '../agent/AgentRunner';
import { Repo } from '../models/Repo.model';

const router = Router();

// 初始化 tools
const tools = new ToolRegistry();
tools.register(readFileTool);
tools.register(listFilesTool);

router.post('/chat', async (req, res) => {
  const { repoId, message } = req.body;

  if (!repoId || !message) {
    return res.status(400).json({ error: 'repoId and message are required' });
  }

  // 查询 repo
  const repo = await Repo.findOne({ repoId });
  if (!repo) {
    return res.status(404).json({ error: 'Repo not found' });
  }

  // 设置 SSE
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const sendEvent = (event: string, data: any) => {
    res.write(`event: ${event}\n`);
    res.write(`data: ${JSON.stringify(data)}\n\n`);
  };

  try {
    const llm = createLLMClient();
    const agent = new AgentRunner(llm, tools);

    await agent.run(message, repoId, repo.name, event => {
      sendEvent(event.type, event.data);
    });

    sendEvent('done', {});
    res.end();
  } catch (error: any) {
    sendEvent('error', { message: error.message });
    res.end();
  }
});

export default router;
```

**在 `src/index.ts` 中注册路由：**
```typescript
import chatRoutes from './api/chat.routes';

app.use('/api', chatRoutes);
```
